# L2 列表库 — 红线清单
# 每行一个测试用例: (expr) → expected
# 测试先写，实现后逐行验证

# Step 19: list primitive + list? predicate
(list) → 0                          # empty list = null
1 (list 1) → pair_count             # single element list
2 (list 1 2 3) → pair_count         # multi-element list
3 (car (list 1 2 3)) → 1
4 (cdr (list 1 2 3)) → pair_count  # (2 3)
5 (list? (list 1 2 3)) → 1         # #t
6 (list? (cons 1 2)) → 0           # #f — dotted pair, not a proper list
7 (list? 42) → 0                   # #f
8 (list? ()) → 1                   # #t — empty list is a proper list

# Step 20: list queries
9 (length (list 1 2 3)) → 3
10 (length ()) → 0
11 (list-ref (list 10 20 30) 0) → 10
12 (list-ref (list 10 20 30) 2) → 30
13 (member 2 (list 1 2 3)) → pair_count  # (2 3)
14 (member 5 (list 1 2 3)) → 0          # #f

# Step 21: list transforms
15 (append (list 1 2) (list 3 4)) → pair_count
16 (reverse (list 1 2 3)) → pair_count  # (3 2 1)
17 (car (reverse (list 1 2 3))) → 3
18 (map (lambda (x) (* x 2)) (list 1 2 3)) → pair_count
19 (car (map (lambda (x) (* x 2)) (list 1 2 3))) → 2
20 (filter (lambda (x) (> x 1)) (list 1 2 3)) → pair_count

# Step 22: equal?
21 (equal? (list 1 2) (list 1 2)) → 1
22 (equal? (list 1 2) (list 1 3)) → 0
23 (equal? "abc" "abc") → 1
24 (equal? "abc" "xyz") → 0
25 (equal? 42 42) → 1
